This is a little test to test the new inline color functionality. And here is also an example plot (btw this uses Olive Green color).

library(omni)
library(ggplot2)
diamonds |> 
    ggplot(aes(carat, price)) +
    geom_hex(color = 'black', linewidth = 0.1) +
    omni::theme_omni() +
    scale_y_continuous(labels = scales::label_dollar()) +
    omni::scale_fill_omni_continuous(palette = 'plum') +
    labs(
        x = 'Diamond Weight (Carat)',
        y = element_blank(),
        title = 'Most diamonds are small {.plum-400 **but as they get heavier**}, 
        they also {.periwinkle-600 *get more expensive*}. **Shocking, I know**. 
        But not as shocking as this title using _automatic line breaks_.',
        subtitle = 'Diamond Price'
    )
